# 注释
useradd hei
echo "root" | passwd --stdin hei &> /dev/null
echo "hei用户创建完成，默认密码是：root"

#!/bin/bash

name=wei

useradd $name
echo "$passwd" | passwd --stdin $name &> /dev/null
echo "用户$name创建完成，默认密码是：$passwd"
